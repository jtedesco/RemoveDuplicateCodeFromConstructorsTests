00064      private static void assertTrue(IResourceDelta expected, IResourceDelta actual) {
00065           assertEqual(expected.getResource(), actual.getResource());
00066           int actualKind= actual.getKind();
00067           int actualFlags= actual.getFlags();
00068           // The real delta can't combine kinds so we remove it from the received one as well.
00069           if ((actualKind & (IResourceDelta.ADDED | IResourceDelta.REMOVED)) != 0) {
00070                actualKind= actualKind & ~IResourceDelta.CHANGED;
00071           }
00072           
00073           // The expected delta doesn't support copy from flag. So remove it
00074           actualFlags= actualFlags & ~IResourceDelta.COPIED_FROM;
00075           
00076           int expectKind= expected.getKind();
00077           int expectedFlags= expected.getFlags() & PRE_DELTA_FLAGS;
00078           if ((expectKind & IResourceDelta.ADDED) != 0 && (expectedFlags & IResourceDelta.MOVED_FROM) != 0) {
00079                expectedFlags= expectedFlags & ~IResourceDelta.OPEN;
00080           }
00081           Assert.assertEquals("Same kind", expectKind, actualKind);
00082           Assert.assertEquals("Same flags", expectedFlags, actualFlags);
00083           IResourceDelta[] expectedChildren=  getExpectedChildren(expected);
00084           IResourceDelta[] actualChildren= getActualChildren(actual, expectedChildren);
00085           Assert.assertEquals("Same number of children", expectedChildren.length, actualChildren.length);
00086           Arrays.sort(expectedChildren, COMPARATOR);
00087           Arrays.sort(actualChildren, COMPARATOR);
00088           for (int i= 0; i < expectedChildren.length; i++) {
00089                assertTrue(expectedChildren[i], actualChildren[i]);
00090           }
00091      }

00094           // This is a simple approach to deal with renamed resources in the deltas.
00095           // However it will not work if there is more than on child per delta since
00096           // the children will be sorted and their order might change.

00101                for(int i= 0; i < expectedPath.segmentCount(); i++) {

00112      private static IResourceDelta[] getExpectedChildren(IResourceDelta delta) {
00113           List result= new ArrayList();
00114           IResourceDelta[] children= delta.getAffectedChildren();
00115           for (int i= 0; i < children.length; i++) {
00116                IResourceDelta child= children[i];
00117                IResource resource= child.getResource();
00118                if (resource != null && isIgnorable(resource))
00119                     continue;
00120                if (child.getAffectedChildren().length > 0) {
00121                     result.add(child);
00122                } else {
00123                     int flags= child.getFlags();
00124                     if (flags == 0 || (flags & PRE_DELTA_FLAGS) != 0) {
00125                          result.add(child);
00126                     }
00127                }
00128           }
00129           return (IResourceDelta[]) result.toArray(new IResourceDelta[result.size()]);
00130      }

00139      private static IResourceDelta[] getActualChildren(IResourceDelta delta, IResourceDelta[] expectedChildren) {
00140           List result= new ArrayList();
00141           if (!IS_COPY_TEST) {
00142                IResourceDelta[] children= delta.getAffectedChildren();
00143                for (int i= 0; i < children.length; i++) {
00144                     IResourceDelta child= children[i];
00145                     IResource resource= child.getResource();
00146                     if (resource != null && isIgnorable(resource))
00147                          continue;
00148                     result.add(child);
00149                }
00150           } else {
00151                IResourceDelta[] candidates= delta.getAffectedChildren();
00152                for (int i= 0; i < candidates.length; i++) {
00153                     IResourceDelta candidate= candidates[i];
00154                     IResource resource= candidate.getResource();
00155                     if (resource != null && isIgnorable(resource))
00156                          continue;
00157                     if (contains(expectedChildren, candidate)) {
00158                          result.add(candidate);
00159                     } else {
00160                          assertCopySource(candidate);
00161                     }
00162                }

00167      private static boolean contains(IResourceDelta[] expectedChildren, IResourceDelta actualDelta) {
00168           IResource actualResource= actualDelta.getResource();

00177           IPath expectedPath= expected.getFullPath();
00178           IPath actualPath= actual.getFullPath();
00179           if (expectedPath.segmentCount()!= actualPath.segmentCount())
00180                return false;
00181           for(int i= 0; i < expectedPath.segmentCount(); i++) {
00182                String expectedSegment= expectedPath.segment(i);
00183                if (expectedSegment.startsWith("UnusedName") || expectedSegment.equals("unusedName"))
00184                     continue;
00185                if (!expectedSegment.equals(actualPath.segment(i)))
00186                     return false;
